## REGRAS OBRIGATÓRIAS
1. Usar `fornecedores_epi` existente — NÃO criar tabela `suppliers`
2. Usar `createAuditLog` em todas as ações
3. Usar `smtpService` para e-mails automáticos
4. Usar `storagePut` para uploads de fotos/documentos
5. Usar `invokeLLM` para IA no mapa de cotação
6. Todo cancelamento de OC faz ROLLBACK FINANCEIRO COMPLETO
7. Confirmar cada passo antes de avançar
8. Rodar `pnpm db:push` após o Passo 1

async function getContaId(db: any, companyId: number, codigo: string) {
  const [conta] = await db.select({ id: financialAccounts.id })
    .from(financialAccounts)
    .where(and(eq(financialAccounts.companyId, companyId), eq(financialAccounts.codigo, codigo)))

  return conta?.id || null;
}

async function getSupplierField(db: any, supplierId: number, campo: string) {

// ── PONTO 1: OC EMITIDA → gera financial_entry + accounts_payable
export async function onOCEmitida(ocId: number, userId: number, userName: string) {
  const db = await getDb();
  if (!db) return;

  const [oc] = await db.select().from(purchaseOrders).where(eq(purchaseOrders.id, ocId));
  if (!oc) return;

  // Mapear tipo da OC para conta financeira
  const codigoConta = oc.tipo === 'servico' ? '3.2' : oc.tipo === 'locacao' ? '3.4' : '3.3';
  const contaId = await getContaId(db, oc.companyId, codigoConta);

  // Criar financial_entry (BLOQUEADO até recebimento)
  const [entryResult] = await db.insert(financialEntries).values({
    companyId: oc.companyId,
    obraId: oc.obraId || null,
    obraNome: oc.obraNome || null,
    contaId,
    tipo: 'despesa',
    natureza: 'variavel',
    valorPrevisto: String(oc.valorTotal || '0'),
    dataCompetencia: new Date().toISOString().split('T')[0],
    dataVencimento: oc.prazoEntrega || null,
    status: 'previsto', // bloqueado até recebimento
    origemModulo: 'compras',
    origemId: oc.id,
    origemDescricao: `OC #${oc.numero || oc.id} — ${oc.supplierNome}`,
    criadoPorId: userId,
    criadoPorNome: userName,
  } as any);

  const financialEntryId = Number(entryResult.insertId);

  // Criar accounts_payable BLOQUEADO
  const [apResult] = await db.insert(accountsPayable).values({
    companyId: oc.companyId,
    ordemId: oc.id,
    supplierId: oc.supplierId,
    supplierNome: oc.supplierNome,
    obraId: oc.obraId || null,
    descricao: `OC #${oc.numero || oc.id} — ${oc.supplierNome}`,
    valorTotal: String(oc.valorTotal || '0'),
    status: 'bloqueado',
    formaPagamento: oc.formaPagamento || null,
    dataVencimento: oc.prazoEntrega || null,
    financialEntryId,
    // Dados bancários copiados automaticamente do fornecedor
    supplierBanco: await getSupplierField(db, oc.supplierId, 'banco'),
    supplierAgencia: await getSupplierField(db, oc.supplierId, 'agencia'),
    supplierConta: await getSupplierField(db, oc.supplierId, 'conta'),
    supplierPix: await getSupplierField(db, oc.supplierId, 'pix'),
    supplierCnpj: await getSupplierField(db, oc.supplierId, 'cnpj'),
  } as any);

  const apId = Number(apResult.insertId);

  // Atualizar OC com vínculos financeiros
  await db.update(purchaseOrders).set({
    financialEntryId,
    accountsPayableId: apId,
  } as any).where(eq(purchaseOrders.id, ocId));

  await createAuditLog({
    userId, userName, action: 'CREATE', module: 'compras',
    entityType: 'oc_lancamento', entityId: ocId,
    details: `OC #${oc.numero} → financial_entry #${financialEntryId} + accounts_payable #${apId}`,
  });
}

// ── PONTO 2: RECEBIMENTO CONFIRMADO → libera pagamento
export async function onRecebimentoConfirmado(
  recebimentoId: number,
  ordemId: number,
  status: 'total' | 'parcial',
  valorLiberado: number,
  userId: number,
  userName: string
) {
  const db = await getDb();
  if (!db) return;

  const [oc] = await db.select().from(purchaseOrders).where(eq(purchaseOrders.id, ordemId));
  if (!oc || !oc.accountsPayableId || !oc.financialEntryId) return;

  if (status === 'total') {
    // Liberar pagamento integral
    await db.update(accountsPayable)
      .set({ status: 'pendente' } as any)
      .where(eq(accountsPayable.id, oc.accountsPayableId));

    await db.update(financialEntries)
      .set({ status: 'a_pagar' } as any)
      .where(eq(financialEntries.id, oc.financialEntryId));

  } else {
    // Liberar proporcionalmente
    const totalOC = parseFloat(String(oc.valorTotal) || '0');
    const percentual = totalOC > 0 ? (valorLiberado / totalOC) : 0;

    await db.update(accountsPayable).set({
      status: 'pendente',
      valorTotal: String(valorLiberado.toFixed(2)),
    } as any).where(eq(accountsPayable.id, oc.accountsPayableId));

    await db.update(financialEntries).set({
      status: 'a_pagar',
      valorPrevisto: String(valorLiberado.toFixed(2)),
      origemDescricao: `Recebimento parcial OC #${oc.numero} (${(percentual * 100).toFixed(1)}%)`,
    } as any).where(eq(financialEntries.id, oc.financialEntryId));
  }

  await createAuditLog({
    userId, userName, action: 'UPDATE', module: 'compras',
    entityType: 'recebimento', entityId: recebimentoId,
    details: `Recebimento ${status} — R$${valorLiberado.toFixed(2)} liberado para pagamento`,
  });
}

// ── PONTO 3: OC CANCELADA → rollback financeiro completo
export async function onOCCancelada(ocId: number, motivo: string, userId: number, userName: string) {
  const db = await getDb();
  if (!db) return;

  const [oc] = await db.select().from(purchaseOrders).where(eq(purchaseOrders.id, ocId));
  if (!oc) return;

  const efeitos: string[] = [];

  // Cancelar financial_entry
  if (oc.financialEntryId) {
    await db.update(financialEntries)
      .set({ status: 'cancelado' } as any)
      .where(eq(financialEntries.id, oc.financialEntryId));
    efeitos.push('financial_entry_cancelado');
  }

  // Cancelar accounts_payable
  if (oc.accountsPayableId) {
    await db.update(accountsPayable)
      .set({ status: 'cancelado' } as any)
      .where(eq(accountsPayable.id, oc.accountsPayableId));
    efeitos.push('accounts_payable_cancelado');
  }

  // Reabrir SC (volta para aprovada)
  if (oc.solicitacaoId) {
    await db.update(purchaseRequests)
      .set({ status: 'aprovada' } as any)
      .where(eq(purchaseRequests.id, oc.solicitacaoId));
    efeitos.push('sc_reaberta');
  }

  // Registrar cancelamento
  await db.insert(purchaseCancellations).values({
    companyId: oc.companyId,
    tipo: 'oc',
    referenciaId: ocId,
    motivo,
    efeitos: JSON.stringify(efeitos),
    canceladoPorId: userId,
    canceladoPorNome: userName,
  } as any);

  await createAuditLog({
    userId, userName, action: 'DELETE', module: 'compras',
    entityType: 'oc_cancelamento', entityId: ocId,
    details: `OC cancelada. Rollback: ${efeitos.join(', ')}. Motivo: ${motivo}`,
  });
}

// ── PONTO 4: COMISSÃO APROVADA → gera pagamento financeiro
export async function onComissaoAprovada(comissaoId: number, userId: number, userName: string) {

  if (!db) return;

  const [comissao] = await db.select().from(buyerCommissions)
    .where(eq(buyerCommissions.id, comissaoId));
  if (!comissao) return;

  const contaId = await getContaId(db, comissao.companyId, '5.3'); // Comissões

  const [entry] = await db.insert(financialEntries).values({
    companyId: comissao.companyId,
    obraId: comissao.obraId,
    obraNome: comissao.obraNome,
    contaId,
    tipo: 'despesa',
    natureza: 'variavel',
    valorPrevisto: String(comissao.valorComissao),
    dataCompetencia: new Date().toISOString().split('T')[0],
    status: 'a_pagar',
    origemModulo: 'comissao_comprador',
    origemId: comissaoId,
    origemDescricao: `Comissão — ${comissao.compradorNome} — ${comissao.obraNome}`,
    criadoPorId: userId,
    criadoPorNome: userName,
  } as any);

  await db.update(buyerCommissions).set({
    financialEntryId: Number(entry.insertId),
    status: 'aprovada_diretor',
    aprovadoPor: userName,
    aprovadoEm: new Date().toISOString(),
  } as any).where(eq(buyerCommissions.id, comissaoId));
}

import { getDb } from "../db";

export function startPurchaseJobs() {

    try {
      await checkOCDeadlines();
      await checkQuotationExpiries();
      await checkContractExpirations();
    } catch(e) { console.error('[PurchaseJobs] Erro:', e); }

  console.log('[PurchaseJobs] Jobs de compras iniciados');

async function checkOCDeadlines() {
  const db = await getDb();
  if (!db) return;
  const hoje = new Date().toISOString().split('T')[0];
  // OCs com prazo vencido sem recebimento → alerta
  const vencidas = await db.select().from(purchaseOrders)
    .where(and(
      eq(purchaseOrders.status, 'emitida'),
      lte(purchaseOrders.prazoEntrega, hoje)
    ));
  for (const oc of vencidas) {
    console.log(`[PurchaseJobs] OC #${oc.numero} prazo vencido: ${oc.prazoEntrega}`);
    // Enviar notificação ao comprador
  }
}

async function checkQuotationExpiries() {
  const db = await getDb();
  if (!db) return;
  const hoje = new Date().toISOString().split('T')[0];
  await db.update(quotations).set({ status: 'expirada' } as any)
    .where(and(eq(quotations.status, 'aberta'), lte(quotations.validadeAte, hoje)));
}

async function checkContractExpirations() {
  const db = await getDb();
  if (!db) return;
  const em7dias = new Date();
  em7dias.setDate(em7dias.getDate() + 7);
  const em7diasStr = em7dias.toISOString().split('T')[0];
  // Alertar contratos vencendo em 7 dias
  const vencendo = await db.select().from(supplierContracts)
    .where(and(eq(supplierContracts.status, 'ativo'), lte(supplierContracts.dataFim, em7diasStr)));
  for (const c of vencendo) {
    if (!c.alertaEnviado) {
      console.log(`[PurchaseJobs] Contrato vencendo: ${c.supplierNome} — ${c.dataFim}`);
      // Enviar notificação
    }
  }
}
